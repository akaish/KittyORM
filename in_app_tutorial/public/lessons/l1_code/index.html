314    void updateExpandPanelList() {
315        if(expandedText != null && expandedLW != null && expandedTextPattern != null) {
316            SimpleDatabase sdb = new SimpleDatabase(getContext());
317            KittyMapper mapper = sdb.getMapper(SimpleExampleModel.class);
318            expandedText.setText(format(expandedTextPattern, mapper.countAll()));
319            List<SimpleExampleModel> models = mapper.findAll();
320            if(models!=null) {
321                LinkedList<String> modelsToString = new LinkedList<>();
322                Iterator<SimpleExampleModel> modelIterator = models.iterator();
323                while (modelIterator.hasNext()) {
324                    modelsToString.addLast(modelIterator.next().toString());
325                }
326                expandedLW.setAdapter(new BasicArrayAdapter(getContext(), modelsToString));
327                expandedLW.setOnTouchListener(new View.OnTouchListener() {
328
329                    // Setting on Touch Listener for handling the touch inside ScrollView

332                        // Disallow the touch request for parent scroll on touch of child view
333                        v.getParent().requestDisallowInterceptTouchEvent(true);
334                        return false;
335                    }
336                });
337            }
338            mapper.close();
339        }
